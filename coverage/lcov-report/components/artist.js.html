'use strict';
 
var React = require('react');
 
/**
 * A component to display an artist.
 *
 * @class Artist
 */
var Artist = React.createClass({
  displayName: 'Artist',
 
  /**
   * @method render
   */
  render: function render() {
    return React.createElement(
      'a',
      { className: 'artist', target: '_blank', href: this.props.url },
      React.createElement('div', { className: 'image',
        style: { backgroundImage: this.constructImageUrl() } }),
      React.createElement(
        'span',
        { className: 'title' },
        this.props.name
      )
    );
  },
 
  /**
   * @method constructImageUrl
   * @return {String} the url required for the artist's image
   */
  constructImageUrl: function constructImageUrl() {
    var url = this.props.image ? 'url(' + this.props.image + ')' : 'none';
    return url;
  }
});
 
module.exports = Artist;
